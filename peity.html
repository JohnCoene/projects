peity

peity is an htmlwidget for Peity : in-line charts .

Install

devtools::install_github("JohnCoene/peity")

Examples

Reproducing examples from the official documentation:

# 1
peity(c(5,3,9,6,5,9,7,3,5,2), type = "bar", options = list(fill = c("red", "green", "blue")))

# 2
peity(c(5,3,2,-1,-3,-2,2,3,5,2), type = "bar", options = list(fill = htmlwidgets::JS("function(value) {
                                                                                     return value > 0 ? 'green':'red'}")))

# 3
fun <- htmlwidgets::JS('function(_, i, all) {
    var g = parseInt((i / all.length) * 255)
    return "rgb(255, " + g + ", 0)"
  }')
peity(c(0,-3,-6,-4,-5,-4,-7,-3,-5,-2), type = "bar", options = list(fill = fun))

# 4
peity(c(4,7,6,5), type = "pie", options = list(fill = c("cyan", "magenta", "yellow", "black")))

# 5
fun2 <- htmlwidgets::JS('function(_, i, all) {
    var g = parseInt((i / all.length) * 255)
    return "rgb(255, " + g + ", 0)"
  }')
peity(c(5,3,9,6,5), type = "pie", options = list(fill = fun2))
