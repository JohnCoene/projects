Chord graph examples

Basic Chord

example

matrix <- matrix(sample(0:1, 100, replace = TRUE, prob = c(0.9,0.6)), nc = 10)

matrix %>% 
  echart(LETTERS[1:10]) %>% 
  echord()

Ribbon Chord

example

matrix <- matrix(sample(0:1, 100, replace = TRUE, prob = c(0.9,0.6)), nc = 10)

matrix %>% 
  echart(LETTERS[1:10]) %>% 
  echord(ribbonType = FALSE)
