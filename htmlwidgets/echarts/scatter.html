Scatter chart examples

Mark points

example

df <- data.frame(x = 1:150,
                 y = round(rnorm(150, mean = 5, sd = 1), 2),
                 z = round(rnorm(150, mean = 7, sd = 1), 2))

pointer <- list(
  show = TRUE,
  type = "cross",
  lineStyle = list(
    type = "dashed",
    width = 0.8
  )
)

df %>% 
  echart(x) %>% 
  escatter(y) %>% 
  escatter(z) %>% 
  emark_point(which = "all", data = list(list(type = "min"), list(type = "max"))) %>% 
  emark_line(which = "all", data = list(list(type = "average"))) %>% 
  etheme("macarons") %>% 
  elegend() %>% 
  etoolbox_full() %>% 
  etooltip(axisPointer = pointer, showDelay = 0)

Bubble

example

df <- data.frame(x = 1:150,
                 y = round(rnorm(150, mean = 5, sd = 1), 2),
                 z = runif(150, 10, 50))

df %>% 
  echart(x) %>% 
  escatter(y, z, name = "scatter 1", symbolSize = 10) %>%
  elegend() %>% 
  etoolbox_full() %>% 
  etoolbox_zoom() %>% 
  exAxis(type = "value") %>% 
  etoolbox_full()

Scale

example

df <- data.frame(x = 1:500, y = runif(500, 1, 20), z = runif(500, 10, 50))

df %>% 
  echart(x) %>% 
  escatter(y, z, symbolSize = 10) %>% 
  ecolorbar(calculable = TRUE, color = list('lightgreen','yellow'), min = 0, max = 5) %>% 
  exAxis(type = "value") %>% 
  etoolbox_full()
