Radar charts

Basic

example

axis <- c("sales", "marketing", "development", "customer support", "IT", "admin")
radar <- data.frame(axis = rep(axis, 2),
                    values = c(6, 16, 30, 38, 52, 25, 7, 13, 33, 25, 46, 37),
                    groups = c(rep("grp1", 6), rep("grp2", 6)))

radar %>% 
  group_by(groups) %>% 
  echart(axis, width = "100%") %>% 
  eradar(values) %>% 
  elegend() %>% 
  etoolbox_full()

Filled

example

axis <- c("sales", "marketing", "development", "customer support", "IT", "admin")
radar <- data.frame(axis = rep(axis, 2),
                    values = c(6, 16, 30, 38, 52, 25, 7, 13, 33, 25, 46, 37),
                    groups = c(rep("grp1", 6), rep("grp2", 6)))

fill <- list(normal = list(areaStyle = list(type = "default")))

radar %>% 
  group_by(groups) %>% 
  echart(axis, width = "100%") %>% 
  eradar(values, itemStyle = fill) %>% 
  elegend() %>% 
  etoolbox_full()
