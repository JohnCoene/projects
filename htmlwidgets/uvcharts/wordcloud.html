Wordcloud

Using dataset from wordcloud2.

library(wordcloud2)
data(demoFreq)

Basic

example

demoFreq$colors <- colorRampPalette(rev(c("lightblue", "yellow", "orange", "red")))(nrow(demoFreq))

demoFreq %>% 
  echart(word) %>% 
  ecloud(freq, 
         colors,
         size = list("100%", "100%"), 
         textPadding = 0, 
         autoSize = list(enable = TRUE,
                         minSize = 5))
